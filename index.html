Spare Parts Inventory Analysis — Complex JOINs & Aggregation

Joins the parts catalog with transaction records and stock levels to identify fast-moving items with critically low inventory, enabling proactive reorder decisions for KOBELCO heavy machinery spare parts.

-- Identify fast-moving spare parts with critically low stock levels
-- Used for inventory reorder prioritization at PT DKCMI (KOBELCO)

SELECT
    p.part_number,
    p.part_name,
    mg.group_name                             AS material_group,
    COUNT(t.transaction_id)                   AS total_transactions,
    SUM(t.quantity)                           AS total_qty_sold,
    ROUND(SUM(t.quantity * t.unit_price), 0)  AS total_revenue,
    ROUND(AVG(t.quantity), 1)                 AS avg_qty_per_order,
    s.current_stock,
    s.reorder_point,
    CASE
        WHEN s.current_stock <= s.reorder_point THEN 'CRITICAL'
        WHEN s.current_stock <= s.reorder_point * 1.5 THEN 'LOW'
        ELSE 'ADEQUATE'
    END                                       AS stock_status
FROM spare_parts p
INNER JOIN transactions t
    ON p.part_id = t.part_id
INNER JOIN material_groups mg
    ON p.group_id = mg.group_id
LEFT JOIN stock_levels s
    ON p.part_id = s.part_id
WHERE t.transaction_date BETWEEN '2023-01-01' AND '2023-09-30'
GROUP BY
    p.part_number, p.part_name,
    mg.group_name, s.current_stock, s.reorder_point
HAVING COUNT(t.transaction_id) >= 10
ORDER BY total_revenue DESC
LIMIT 20;

LPG Monthly Trend Analysis — Window Functions

Uses window functions to calculate month-over-month growth rates and running totals for LPG distribution, enabling trend analysis and seasonal demand pattern identification across 2023–2024.

-- Month-over-month growth analysis for LPG 3Kg distribution
-- Compares 2023 vs 2024 with running totals and growth rates

WITH monthly_data AS (
    SELECT
        EXTRACT(YEAR FROM sale_date)   AS sale_year,
        EXTRACT(MONTH FROM sale_date)  AS sale_month,
        customer_segment,
        COUNT(*)                       AS num_transactions,
        SUM(quantity)                  AS total_volume,
        SUM(profit)                   AS total_profit
    FROM lpg_distributions
    WHERE sale_date BETWEEN '2023-01-01' AND '2024-12-31'
    GROUP BY
        EXTRACT(YEAR FROM sale_date),
        EXTRACT(MONTH FROM sale_date),
        customer_segment
)
SELECT
    sale_year,
    sale_month,
    customer_segment,
    total_volume,
    total_profit,
    SUM(total_volume) OVER (
        PARTITION BY sale_year, customer_segment
        ORDER BY sale_month
    )                                  AS running_volume,
    LAG(total_volume) OVER (
        PARTITION BY customer_segment, sale_month
        ORDER BY sale_year
    )                                  AS prev_year_volume,
    ROUND(
        (total_volume - LAG(total_volume) OVER (
            PARTITION BY customer_segment, sale_month
            ORDER BY sale_year
        )) * 100.0 / NULLIF(LAG(total_volume) OVER (
            PARTITION BY customer_segment, sale_month
            ORDER BY sale_year
        ), 0), 1
    )                                  AS yoy_growth_pct
FROM monthly_data
ORDER BY customer_segment, sale_year, sale_month;

Restaurant Top Customer Analysis — CTE with Ranking

Uses CTEs and ranking functions to identify the most valuable customers by revenue contribution, transaction frequency, and average order value — supporting targeted marketing and loyalty program strategies.

-- Top customer analysis with RFM-style ranking
-- Supports customer segmentation for marketing at PT JAR Andalan Rasa

WITH customer_metrics AS (
    SELECT
        c.customer_id,
        c.customer_name,
        c.channel,
        COUNT(DISTINCT o.order_id)             AS total_orders,
        SUM(oi.quantity * oi.unit_price)        AS total_spent,
        ROUND(AVG(oi.quantity * oi.unit_price), 0) AS avg_order_value,
        MAX(o.order_date)                       AS last_order_date,
        DATEDIFF(CURRENT_DATE, MAX(o.order_date)) AS days_since_last
    FROM customers c
    INNER JOIN orders o ON c.customer_id = o.customer_id
    INNER JOIN order_items oi ON o.order_id = oi.order_id
    WHERE o.order_date >= '2025-01-01'
    GROUP BY c.customer_id, c.customer_name, c.channel
),
ranked_customers AS (
    SELECT
        *,
        DENSE_RANK() OVER (ORDER BY total_spent DESC)   AS spend_rank,
        DENSE_RANK() OVER (ORDER BY total_orders DESC)   AS frequency_rank,
        NTILE(4) OVER (ORDER BY total_spent DESC)        AS spend_quartile,
        ROUND(total_spent * 100.0 / SUM(total_spent) OVER (), 2)
                                                          AS revenue_share_pct
    FROM customer_metrics
)
SELECT
    customer_name,
    channel,
    total_orders,
    total_spent,
    avg_order_value,
    days_since_last,
    spend_rank,
    revenue_share_pct,
    CASE spend_quartile
        WHEN 1 THEN 'VIP'
        WHEN 2 THEN 'Regular'
        WHEN 3 THEN 'Occasional'
        ELSE 'At Risk'
    END AS customer_tier
FROM ranked_customers
ORDER BY spend_rank
LIMIT 25;

Restaurant Data Cleaning Pipeline — Pandas

A complete data cleaning pipeline for restaurant transaction data: loading raw exports, handling missing values, standardizing formats, removing duplicates, and engineering features for analysis.

import pandas as pd
import numpy as np

# Load raw restaurant transaction data
df = pd.read_csv('restaurant_transactions_raw.csv', parse_dates=['order_date'])
print(f"Raw data: {df.shape[0]:,} rows, {df.shape[1]} columns")

# ---- Data Cleaning Pipeline ----

# 1. Handle missing values
df['payment_method'].fillna('Unknown', inplace=True)
df['channel'].fillna('Dine-in', inplace=True)
df.dropna(subset=['menu_item', 'quantity', 'unit_price'], inplace=True)

# 2. Remove duplicates
before = len(df)
df.drop_duplicates(subset=['order_id', 'menu_item', 'order_date'], inplace=True)
print(f"Removed {before - len(df)} duplicate rows")

# 3. Standardize text columns
df['menu_item'] = df['menu_item'].str.strip().str.title()
df['category'] = df['category'].str.strip().str.upper()

# 4. Fix data types and outliers
df['quantity'] = df['quantity'].astype(int)
df = df[df['unit_price'] > 0]  # Remove zero/negative prices

# 5. Feature engineering
df['total_amount'] = df['quantity'] * df['unit_price']
df['order_hour'] = df['order_date'].dt.hour
df['day_of_week'] = df['order_date'].dt.day_name()
df['is_weekend'] = df['order_date'].dt.dayofweek.isin([5, 6])
df['month'] = df['order_date'].dt.to_period('M')

# 6. Validation summary
print(f"\nCleaned data: {df.shape[0]:,} rows")
print(f"Date range: {df['order_date'].min()} to {df['order_date'].max()}")
print(f"Total revenue: Rp {df['total_amount'].sum():,.0f}")
print(f"Unique menu items: {df['menu_item'].nunique()}")

LPG Distribution Visualization — Matplotlib & Seaborn

Creates a comprehensive 2x2 dashboard visualizing LPG distribution metrics: monthly volume trends with YoY comparison, customer segmentation breakdown, profit margins, and seasonal heatmap.

import matplotlib.pyplot as plt
import seaborn as sns
import pandas as pd
import numpy as np

# Load cleaned LPG distribution data
df = pd.read_csv('lpg_distribution_clean.csv', parse_dates=['sale_date'])

# Set theme
sns.set_theme(style='whitegrid', palette='viridis')
fig, axes = plt.subplots(2, 2, figsize=(16, 12))
fig.suptitle('LPG 3Kg Distribution Dashboard — 2023 vs 2024', fontsize=18, fontweight='bold')

# --- Chart 1: Monthly Volume YoY Comparison ---
monthly = df.groupby([df['sale_date'].dt.year.rename('year'),
                       df['sale_date'].dt.month.rename('month')])['quantity'].sum().reset_index()
for year in [2023, 2024]:
    data = monthly[monthly['year'] == year]
    axes[0, 0].plot(data['month'], data['quantity'], marker='o', linewidth=2.5, label=str(year))
axes[0, 0].set_title('Monthly Distribution Volume', fontweight='bold')
axes[0, 0].set_xlabel('Month')
axes[0, 0].set_ylabel('Quantity (units)')
axes[0, 0].legend()

# --- Chart 2: Customer Segment Breakdown ---
segment = df.groupby('customer_segment')['quantity'].sum()
colors = ['#0ea5e9', '#8b5cf6']
axes[0, 1].pie(segment.values, labels=segment.index, autopct='%1.1f%%',
               colors=colors, startangle=90, textprops={'fontsize': 12})
axes[0, 1].set_title('Customer Segmentation', fontweight='bold')

# --- Chart 3: Monthly Profit Comparison ---
profit_monthly = df.groupby([df['sale_date'].dt.year.rename('year'),
                              df['sale_date'].dt.month.rename('month')])['profit'].sum().reset_index()
pivot = profit_monthly.pivot(index='month', columns='year', values='profit')
pivot.plot(kind='bar', ax=axes[1, 0], width=0.7)
axes[1, 0].set_title('Monthly Profit Comparison (Rp)', fontweight='bold')
axes[1, 0].set_xlabel('Month')
axes[1, 0].tick_params(axis='x', rotation=0)

# --- Chart 4: Weekly Demand Heatmap ---
df['day_of_week'] = df['sale_date'].dt.day_name()
df['week_num'] = df['sale_date'].dt.isocalendar().week
heatmap_data = df.pivot_table(index='day_of_week', columns='week_num', values='quantity', aggfunc='sum')
sns.heatmap(heatmap_data.iloc[:, :12], ax=axes[1, 1], cmap='YlOrRd', fmt='.0f')
axes[1, 1].set_title('Weekly Demand Heatmap (Q1)', fontweight='bold')

plt.tight_layout()
plt.savefig('lpg_dashboard.png', dpi=150, bbox_inches='tight')
plt.show()

Spare Parts Statistical Summary — NumPy & Pandas

Generates a comprehensive statistical analysis of spare parts transactions, including ABC classification, reorder point calculations, and demand variability metrics for inventory optimization.

import pandas as pd
import numpy as np

# Load spare parts transaction data
df = pd.read_csv('spare_parts_transactions.csv', parse_dates=['transaction_date'])

# ---- Statistical Summary ----
print("=== SPARE PARTS ANALYTICS SUMMARY ===")
print(f"Period: {df['transaction_date'].min():%b %Y} to {df['transaction_date'].max():%b %Y}")
print(f"Total Transactions: {len(df):,}")
print(f"Total Revenue: Rp {df['total_price'].sum():,.0f}")
print(f"Unique Parts: {df['part_number'].nunique()}")

# ---- ABC Classification ----
part_revenue = df.groupby('part_number')['total_price'].sum().sort_values(ascending=False)
cumulative_pct = part_revenue.cumsum() / part_revenue.sum() * 100

abc_class = np.where(cumulative_pct <= 80, 'A', np.where(cumulative_pct <= 95, 'B', 'C'))
abc_df = pd.DataFrame({'revenue': part_revenue, 'cum_pct': cumulative_pct, 'class': abc_class})

print("\n=== ABC CLASSIFICATION ===")
for cls in ['A', 'B', 'C']:
    subset = abc_df[abc_df['class'] == cls]
    print(f"Class {cls}: {len(subset)} parts | Rp {subset['revenue'].sum():,.0f} "
          f"({subset['revenue'].sum()/part_revenue.sum()*100:.1f}%)")

# ---- Demand Variability & Reorder Points ----
monthly = df.groupby(['part_number', df['transaction_date'].dt.to_period('M')])['quantity'].sum()
demand_stats = monthly.groupby('part_number').agg(['mean', 'std']).round(1)
demand_stats.columns = ['avg_monthly_demand', 'std_demand']
demand_stats['cv'] = (demand_stats['std_demand'] / demand_stats['avg_monthly_demand']).round(2)

# Safety stock = Z * sigma * sqrt(lead_time), Z=1.65 for 95% service level
lead_time_months = 0.5
demand_stats['safety_stock'] = np.ceil(
    1.65 * demand_stats['std_demand'] * np.sqrt(lead_time_months)
)
demand_stats['reorder_point'] = np.ceil(
    demand_stats['avg_monthly_demand'] * lead_time_months + demand_stats['safety_stock']
)

print("\n=== TOP 10 FAST-MOVING PARTS (REORDER ANALYSIS) ===")
top10 = demand_stats.nlargest(10, 'avg_monthly_demand')
print(top10[['avg_monthly_demand', 'cv', 'safety_stock', 'reorder_point']].to_string())
